\newcommand{\myage}[3]{%
    \newcounter{age}
    \setcounter{age}{\the\year}
    \addtocounter{age}{-#1}
    \ifthenelse{\the\month < #2}{
        \addtocounter{age}{-1}
    }{
        \ifthenelse{\the\month = #2}{
            \ifthenelse{\the\day < #3}{
                \addtocounter{age}{-1}
            }{}
        }{}
    }
    \theage
}
